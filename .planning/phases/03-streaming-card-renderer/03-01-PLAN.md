    ```python
    class CardStreamingManager:
        def __init__(self, card_id: str, seq_id: str, http: httpx.AsyncClient, api_client):
            self.card_id = card_id
            self.seq_id = seq_id
            self._http = http
            self._api_client = api_client
            self._buffer: list[str] = []          # accumulated text chunks
            self._tool_sections: list[str] = []   # formatted tool markdown sections
            self._sequence_created = False
            self._flush_task: asyncio.Task | None = None
            self._dirty = False                   # True when buffer has unflushed content
            self._logger = structlog.get_logger()

        def append_text(self, text: str) -> None:
            """Append a text chunk to the buffer (from TextBlock). Per D-21: does NOT trigger immediate flush."""
            self._buffer.append(text)
            self._dirty = True

        def append_tool_use(self, block) -> None:
            """Append tool use section (from ToolUseBlock). Per D-24/D-25: shown before text."""
            # Format: blockquote with tool name and brief summary
            name = getattr(block, 'name', 'unknown')
            input_data = getattr(block, 'input', {})
            # Extract a brief summary from input (first 80 chars of command or description)
            summary = ""
            if isinstance(input_data, dict):
                for key in ('command', 'content', 'file_path', 'pattern', 'description'):
                    if key in input_data:
                        val = str(input_data[key])
                        summary = val[:80] + ("..." if len(val) > 80 else "")
                        break
            section = f"\n> **{name}**"
            if summary:
                section += f": `{summary}`"
            section += "\n"
            self._tool_sections.append(section)
            self._dirty = True

        def append_tool_result(self, block) -> None:
            """Append tool result section (from ToolResultBlock). Per D-06/CARD-06."""
            content = getattr(block, 'content', None)
            is_error = getattr(block, 'is_error', False)
            if content:
                text = str(content)[:100]  # Truncate long results
                prefix = "Error" if is_error else "Result"
                section = f"> _{prefix}: {text}_\n"
                self._tool_sections.append(section)
                self._dirty = True

        def _build_content(self, include_typing: bool = True) -> str:
            """Build full card content from tool sections + text buffer."""
            parts = self._tool_sections + self._buffer
            content = "".join(parts)
            if include_typing:
                content += "\n\n_正在输入..._"
            return content

        async def _flush_to_card(self) -> None:
            """Flush current buffer to CardKit sequence API. Per D-21: called at 300-500ms intervals."""
            if not self._dirty:
                return
            content = self._build_content(include_typing=True)
            try:
                if not self._sequence_created:
                    await create_sequence(self._http, self._api_client, self.card_id, self.seq_id, content)
                    self._sequence_created = True
                else:
                    await _update_sequence_with_retry(self._http, self._api_client, self.card_id, self.seq_id, content)
                self._dirty = False
            except Exception as exc:
                self._logger.error("card_flush_error", card_id=self.card_id, error=str(exc))

        async def _flush_loop(self) -> None:
            """Background loop: flush every 400ms (middle of 300-500ms range per D-21)."""
            try:
                while True:
                    await asyncio.sleep(0.4)
                    await self._flush_to_card()
            except asyncio.CancelledError:
                pass  # Normal shutdown

        def start_flush_loop(self) -> None:
            """Start the background flush loop as an asyncio task."""
            if self._flush_task is None:
                self._flush_task = asyncio.create_task(self._flush_loop())

        async def stop_flush_loop(self) -> None:
            """Stop the background flush loop."""
            if self._flush_task is not None:
                self._flush_task.cancel()
                try:
                    await self._flush_task
                except asyncio.CancelledError:
                    pass
                self._flush_task = None

        async def finalize(self) -> None:
            """Final flush without typing indicator + finish sequence. Per D-27: complete response only."""
            await self.stop_flush_loop()
            # Final content without typing indicator
            final_content = self._build_content(include_typing=False)
            try:
                if not self._sequence_created:
                    await create_sequence(self._http, self._api_client, self.card_id, self.seq_id, final_content)
                else:
                    await _update_sequence_with_retry(self._http, self._api_client, self.card_id, self.seq_id, final_content)
            except Exception as exc:
                self._logger.error("card_finalize_flush_error", card_id=self.card_id, error=str(exc))
            # ALWAYS finish sequence — per Pitfall 4: leaving sequence open causes infinite animation
            await finish_sequence(self._http, self._api_client, self.card_id, self.seq_id)
    ```

    async def create_streaming_card(client: lark.Client) -> str:
        """Create a CardKit streaming card. Returns card_id for sequence API calls.

        Per D-19/D-20: Creates the streaming card resource on the server.
        The caller must then call patch_im_with_card_id() to link it to the IM message.
        """
        from lark_oapi.api.cardkit.v1 import CreateCardRequest, CreateCardRequestBody

        streaming_card_data = {
            "schema": "2.0",
            "config": {
                "streaming_mode": True,
                "streaming_config": {
                    "print_frequency_ms": {"default": 70},
                    "print_step": {"default": 1},
                    "print_strategy": "fast",
                },
            },
            "header": {
                "title": {"tag": "plain_text", "content": "AI 助手"},
                "template": "blue",
            },
            "body": {
                "elements": [{"tag": "markdown", "content": "_正在输入..._"}]
            },
        }

        request = (
            CreateCardRequest.builder()
            .request_body(
                CreateCardRequestBody.builder()
                .type("card_json")
                .data(json.dumps(streaming_card_data, ensure_ascii=False))
                .build()
            )
            .build()
        )
        resp = await client.cardkit.v1.card.acreate(request)
        if not resp.success():
            raise RuntimeError(f"CardKit create failed: {resp.code} {resp.msg}")

        logger.debug("streaming_card_created", card_id=resp.data.card_id)
        return resp.data.card_id
    ```

    async def patch_im_with_card_id(client: lark.Client, message_id: str, card_id: str) -> None:
        """Patch IM message to embed a CardKit card_id. Per D-19/Pitfall 3/Pitfall 5."""
        card_id_content = json.dumps(
            {"type": "card", "data": {"card_id": card_id}},
            ensure_ascii=False,
        )

        request = (
            lark.im.v1.PatchMessageRequest.builder()
            .message_id(message_id)
            .request_body(
                lark.im.v1.PatchMessageRequestBody.builder()
                .content(card_id_content)
                .build()
            )
            .build()
        )
        resp = await client.im.v1.message.apatch(request)
        if not resp.success():
            raise RuntimeError(f"CardKit IM patch failed: {resp.code} {resp.msg}")

        logger.debug("im_patched_with_card_id", message_id=message_id, card_id=card_id)
    ```
